#[no_mangle]
pub unsafe extern "C" fn json_object_deep_copy(
    mut src: *mut crate::json_object_private_h::json_object,
    mut dst: *mut *mut crate::json_object_private_h::json_object,
    mut shallow_copy: Option<crate::src::json_object::json_c_shallow_copy_fn>,
) -> ::core::ffi::c_int {
    let mut rc: ::core::ffi::c_int = 0;

    /* Check if arguments are sane ; *dst must not point to a non-NULL object */
    if src.is_null() || dst.is_null() || !(*dst).is_null() {
        *crate::stdlib::__errno_location() = crate::stdlib::EINVAL;
        return -(1 as ::core::ffi::c_int);
    }
    if shallow_copy.is_none() {
        shallow_copy = Some(
            json_c_shallow_copy_default
                as unsafe extern "C" fn(
                    *mut crate::json_object_private_h::json_object,
                    *mut crate::json_object_private_h::json_object,
                    *const ::core::ffi::c_char,
                    crate::__stddef_size_t_h::size_t,
                    *mut *mut crate::json_object_private_h::json_object,
                ) -> ::core::ffi::c_int,
        );
    }
    rc = json_object_deep_copy_recursive(
        src,
        ::core::ptr::null_mut::<crate::json_object_private_h::json_object>(),
        ::core::ptr::null::<::core::ffi::c_char>(),
        -(1 as ::core::ffi::c_int) as crate::__stddef_size_t_h::size_t,
        dst,
        shallow_copy,
    );
    if rc < 0 as ::core::ffi::c_int {
        json_object_put(*dst);
        *dst = ::core::ptr::null_mut::<crate::json_object_private_h::json_object>();
    }
    return rc;
}
