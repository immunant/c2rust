#[no_mangle]
pub unsafe extern "C" fn json_object_iter_peek_value(
    mut iter: *const crate::src::json_object_iterator::json_object_iterator,
) -> *mut json_object {
    return (*((*iter).opaque_ as *const lh_entry)).v as uintptr_t as *mut ::core::ffi::c_void
        as *mut json_object;
}
