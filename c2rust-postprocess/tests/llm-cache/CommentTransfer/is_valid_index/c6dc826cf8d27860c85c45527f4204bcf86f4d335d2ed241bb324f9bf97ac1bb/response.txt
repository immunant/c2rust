unsafe extern "C" fn is_valid_index(
    mut jo: *mut crate::json_object_private_h::json_object,
    mut path: *const ::core::ffi::c_char,
    mut idx: *mut crate::stdlib::int32_t,
) -> ::core::ffi::c_int {
    let mut i: ::core::ffi::c_int = 0;
    let mut len: ::core::ffi::c_int = crate::stdlib::strlen(path) as ::core::ffi::c_int;
    if len == 1 as ::core::ffi::c_int {
        /* this code-path optimizes a bit, for when we reference the 0-9 index range in a JSON array
           and because leading zeros not allowed */
        if *(*crate::stdlib::__ctype_b_loc()).offset(*path.offset(0 as ::core::ffi::c_int as isize)
            as ::core::ffi::c_uchar as ::core::ffi::c_int
            as isize) as ::core::ffi::c_int
            & crate::stdlib::_ISdigit as ::core::ffi::c_int as ::core::ffi::c_ushort as ::core::ffi::c_int
            != 0
        {
            *idx = (*path.offset(0 as ::core::ffi::c_int as isize) as ::core::ffi::c_int
                - '0' as i32) as crate::stdlib::int32_t;
        } else {
            *crate::stdlib::__errno_location() = crate::stdlib::EINVAL;
            return 0 as ::core::ffi::c_int;
        }
    } else {
        /* leading zeros not allowed per RFC */
        if *path.offset(0 as ::core::ffi::c_int as isize) as ::core::ffi::c_int == '0' as i32 {
            *crate::stdlib::__errno_location() = crate::stdlib::EINVAL;
            return 0 as ::core::ffi::c_int;
        }
        /* RFC states base-10 decimals */
        i = 0 as ::core::ffi::c_int;
        while i < len {
            if *(*crate::stdlib::__ctype_b_loc()).offset(
                *path.offset(i as isize) as ::core::ffi::c_uchar as ::core::ffi::c_int as isize
            ) as ::core::ffi::c_int
                & crate::stdlib::_ISdigit as ::core::ffi::c_int as ::core::ffi::c_ushort as ::core::ffi::c_int
                == 0
            {
                *crate::stdlib::__errno_location() = crate::stdlib::EINVAL;
                return 0 as ::core::ffi::c_int;
            }
            i += 1;
        }
        *idx = crate::stdlib::strtol(
            path,
            ::core::ptr::null_mut::<*mut ::core::ffi::c_char>(),
            10 as ::core::ffi::c_int,
        ) as crate::stdlib::int32_t;
        if *idx < 0 as crate::stdlib::int32_t {
            *crate::stdlib::__errno_location() = crate::stdlib::EINVAL;
            return 0 as ::core::ffi::c_int;
        }
    }
    len = crate::src::json_object::json_object_array_length(jo) as ::core::ffi::c_int;
    if *idx >= len as crate::stdlib::int32_t {
        *crate::stdlib::__errno_location() = crate::stdlib::ENOENT;
        return 0 as ::core::ffi::c_int;
    }
    return 1 as ::core::ffi::c_int;
}
